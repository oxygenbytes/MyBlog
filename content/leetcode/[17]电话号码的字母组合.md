class Solution {
public:
    vector<string> dict{"", "", "abc", "def", "ghi", "jkl", "mno", "pqrs", "tuv", "wxyz"};
    vector<string> ans;
    string cur;
    vector<string> letterCombinations(string digits) {
        if(!digits.size()) return ans;
        dfs(digits, 0);
        return ans;
    }

    void dfs(string& digits, int level){
        if(level == digits.size())
        {
            ans.push_back(cur);
            return ;
        }

        auto str = dict[digits[level] - '0'];
        for(int i = 0;i < str.size();i++){
            string temp = cur;
            cur += str[i];
            dfs(digits, level + 1);
            cur = temp;
        }
    }
};
