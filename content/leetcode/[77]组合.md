#include "leetcode.h"

class Solution {
public:
    vector<vector<int>> ans;
    vector<int> cur;
    vector<vector<int>> combine(int n, int k) {
        if(k > n || k <= 0) return ans;
        dfs(n, k, 1);
        return ans;
    }
    void dfs(int n, int k, int level){
        if(cur.size() == k){
            ans.push_back(cur);
            return ;
        }
        if(n - level + 1 < k - cur.size()) return ; // å‰ªæ
        for(int i = level;i <= n;i++){
            cur.push_back(i);
            dfs(n, k, i+1);
            cur.pop_back();
        }
    }
};
