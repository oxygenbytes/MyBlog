class Solution {
public:
    int majorityElement(vector<int>& nums) {

        if(nums.size() == 0) return 0;
        if(nums.size() == 1) return nums[0];

        int ans = nums[0];
        int freq = 1;
        for(int i = 1;i < nums.size();i++){
            if(!freq) ans = nums[i], freq++;
            else{
                if(ans == nums[i]) freq++;
                else freq--;
            }
        }

        return ans;
    }
};
