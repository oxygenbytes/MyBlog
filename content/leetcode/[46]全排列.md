#include "leetcode.h"

class Solution {
public:
    vector<vector<int>> ans;
    vector<int> cur;
    vector<vector<int>> permute(vector<int>& nums) {
        int n = nums.size();
        vector<bool> vis(n + 1, false);
        dfs(n, nums, vis);
        return ans;
    }

    void dfs(int n, vector<int>& nums, vector<bool>& vis){
        if(cur.size() == n){
            ans.push_back(cur);
            return ;
        }

        for(int i = 1;i <= n;i++){
            if(!vis[i]){
                cur.push_back(nums[i-1]);
                vis[i] = true;
                dfs(n, nums, vis);
                vis[i] = false;
                cur.pop_back();
            }
        }
    }
};
