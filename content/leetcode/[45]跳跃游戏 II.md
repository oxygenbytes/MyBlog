class Solution {
public:
    int jump(vector<int>& nums) {
        if(nums.size() == 1) return 0;

        int l = 0, r = 0, step = 0;
        while(l <= r){
            int max_r = 0;
            for(int i = l;i <= r;i++){
                max_r = max(max_r, i + nums[i]);
            }
            l = r + 1, r = max_r;
            step++;
            // cout<<0u - 1<<endl;  4294967295
            // cout<<(int)(0u-1)<<endl;
            if(r >= (int)nums.size() - 1) break; // é»˜è®¤è½¬å‹ int -> unsigned int
        }
        return step;
    }
};
