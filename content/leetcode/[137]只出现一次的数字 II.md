class Solution {
public:
    int singleNumber(vector<int>& nums) {
        long long int ans = 0;
        for(int i = 31;i >= 0;i--){
            int cnt = 0;
            for(auto num : nums) cnt += num >> i & 1;
            if(cnt % 3 == 1) ans = ans * 2 + 1;
            if(cnt % 3 == 0) ans = ans * 2;
        }
        return ans;
    }
};
