class Solution {
public:
    vector<int> spiralOrder(vector<vector<int>>& matrix) {
        if(matrix.empty() || matrix[0].empty()) return {};
        int m = matrix.size();
        int n = matrix[0].size();

        vector<int> ans;
        int d = 1;
        int i = 0, j = 0;
        int dx[4] = {-1, 0, 1, 0};
        int dy[4] = {0, 1, 0, -1};
        for(int k = 0;k < m * n;k++){
            ans.push_back(matrix[i][j]);
            matrix[i][j] = 0;
            int a = i + dx[d];
            int b = j + dy[d];
            if(a < 0 || a >= m || b < 0 || b >= n || matrix[a][b] == 0){
                d = (d + 1) % 4;
                a = i + dx[d];
                b = j + dy[d];
            }
            i = a, j = b;
        }

        return ans;
    }
};
