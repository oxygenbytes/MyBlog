class Solution {
public:
    int trap(vector<int>& height) {
        if(height.empty()) return 0;
        int n = height.size();
        int left = 0, right = n - 1;
        int ans = 0;

        int l_max = height[0]; // height[0,,left] æœ€é«˜æŸ±å­
        int r_max = height[n-1]; // height[right,end] æœ€é«˜æŸ±å­

        while(left <= right){
            l_max = max(l_max, height[left]);
            r_max = max(r_max, height[right]);

            if(l_max < r_max){ // æ­¤æ—¶[right,end]çš„é«˜åº¦ä¸æ˜¯çŸ­æ¿ï¼Œä¸å½±å“leftå¤„æ°´çš„é«˜åº¦
                ans += l_max - height[left];
                left++;
            }else{
                ans += r_max - height[right];
                right--;
            }
        }
        return ans;
    }
};
