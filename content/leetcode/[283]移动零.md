class Solution {
public:
    void moveZeroes(vector<int>& nums) {
        int n = nums.size();
        int j = 0; // mark end of non-zero seq
        for(int i = 0;i < n;i++){
            if(nums[i]){
                swap(nums[i], nums[j]);
                j++;
            }
        }
    }
};
