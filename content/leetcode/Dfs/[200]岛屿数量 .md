#include "leetcode.h" 
//leetcode submit region begin(Prohibit modification and deletion)
class Solution {
public:
    int numIslands(vector<vector<char>>& grid) {
        int m = grid.size();
        int n = grid[0].size();
        int res = 0;
        for(int i = 0;i < m;i++){
            for(int j = 0;j < n;j++){
                if(grid[i][j] == '1'){
                    dfs(grid, i, j);
                    res++;
                }
            }
        }
        return res;
    }
    int dx[4] = {-1, 0, 1, 0};
    int dy[4] = {0, 1, 0, -1};
    void dfs(vector<vector<char>>& grid, int x,int y){
        grid[x][y] = '0';
        for(int i = 0;i < 4;i++){
            int a = dx[i] + x;
            int b = dy[i] + y;
            if(a < 0 || a >= grid.size() || b < 0 || b >= grid[0].size() || grid[a][b] == '0')
                continue;
            dfs(grid, a, b);
        }
    }
};
