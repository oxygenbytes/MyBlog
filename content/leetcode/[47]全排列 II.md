class Solution {
public:
    vector<vector<int>> ans;
    vector<int> path;
    vector<bool> vis;
    vector<vector<int>> permuteUnique(vector<int>& nums) {
        int n = nums.size();
        sort(nums.begin(), nums.end());
        vis = vector<bool>(n, false);

        dfs(nums, 0);
        return ans;
    }

    void dfs(const vector<int>& nums, int cur){
        if(cur == nums.size()){
            ans.push_back(path);
            return ;
        }
// ä¾‹å¦‚[1 2 2â€˜]å¯èƒ½å‡ºç°[1 2 2'] å’Œ[1 2â€˜ 2] çš„æƒ…å†µ å¦‚æœâ€œå­˜åœ¨å‰ä¸€ä¸ªç›¸åŒå…ƒç´ â€ ä¸”â€œæœªè¢«ä½¿ç”¨è¿‡â€, å½“ç°æœ‰æ’åˆ—æ˜¯[1 2']æ—¶
// åŸæ¥çš„æ•°ç»„[1 2 2â€˜]ä¸­2â€™å­˜åœ¨å‰ä¸€ä¸ªå…ƒç´ 2ä¸å…¶ç›¸åŒ, ä¸”æ­¤æ—¶2æœªè¢«è®¿é—®è¿‡ï¼Œè·³è¿‡ã€‚[1 2 2']çš„æ’åˆ—å…ˆäº[1 2' 2]å­˜åœ¨ï¼Œå› æ­¤å¯ä»¥å»é™¤ã€‚
        for(int i = 0;i < nums.size();i++){
            if(!vis[i]){

                vis[i] = true;
                path.push_back(nums[i]);
                dfs(nums, cur + 1);
                path.pop_back();
                vis[i] = false;
                // [2,2',3] if cur(0) == 2, then skip cur(1)
                while(i + 1< nums.size() && nums[i] == nums[i+1]) i++; // here is while
            }
        }
    }
};
