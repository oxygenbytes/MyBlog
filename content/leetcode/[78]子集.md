#include "leetcode.h"

class Solution {
public:
    vector<vector<int>> ans;
    vector<int> cur;
    vector<vector<int>> subsets(vector<int>& nums) {
        dfs(0, nums);
        return ans;
    }
    // å­é›†ã€ç»„åˆç±»é—®é¢˜ï¼Œæ— å…³é¡ºåºï¼Œå…³é”®æ˜¯ç”¨ä¸€ä¸ª start å‚æ•°æ¥å¯¹æ•°ç»„è¿›è¡Œç­›é€‰
    void dfs(int start, vector<int>& nums){
        ans.push_back(cur);
        for(int i = start;i < nums.size();i++){
            cur.push_back(nums[i]);
            dfs(i+1, nums);
            cur.pop_back();
        }
    }
};
